        // ── FILE HANDLER ──
        async function handleFile(file) {
            const isImg = file.type.startsWith('image/');
            const isPdf = file.type === 'application/pdf';
            if (!isImg && !isPdf) return alert('Invalid file format. Please upload Image or PDF.');
            currentFile = file;
            if (isPdf) {
                showLoading("Reading PDF...", "Extracting secure image");
                try {
                    const buf = await file.arrayBuffer();
                    const pdf = await pdfjsLib.getDocument(buf).promise;
                    const page = await pdf.getPage(1);
                    const vp = page.getViewport({ scale: 2.0 });
                    const cvs = document.createElement('canvas');
                    cvs.width = vp.width; cvs.height = vp.height;
                    await page.render({ canvasContext: cvs.getContext('2d'), viewport: vp }).promise;
                    cvs.toBlob(blob => {
                        currentFile = new File([blob], file.name.replace('.pdf', '.png'), { type: 'image/png' });
                        setupPreview(currentFile, URL.createObjectURL(blob));
                        hideLoading();
                    });
                } catch (e) { hideLoading(); alert("Failed to read PDF."); }
            } else {
                const reader = new FileReader();
                reader.onload = e => setupPreview(file, e.target.result);
                reader.readAsDataURL(file);
            }
        }

        // ── BASE64 ENCODER ──
        function encodeBase64(str) {
            if (!str) return '';
            return btoa(encodeURIComponent(str).replace(/%([0-9A-F]{2})/g, function(match, p1) {
                return String.fromCharCode('0x' + p1);
            }));
        }

        // ── PROCESS (CONVERT/COMPRESS) ──
        async function process(actionType) {
            if (!currentFile) return alert('Please select a file first!');
            showLoading(actionType === 'convert' ? 'Converting...' : 'Compressing...', 'Working magic on your file');
            const formData = new FormData();
            formData.append('image', currentFile);
            formData.append('format', selectedFormat);
            if (actionType === 'compress') {
                formData.append('targetSize', document.getElementById('target').value);
                formData.append('sizeUnit', document.getElementById('unit').value);
                formData.append('width', document.getElementById('w').value);
                formData.append('height', document.getElementById('h').value);
            }
            try {
                const response = await fetch(`/${actionType}`, { method: 'POST', body: formData });
                const data = await response.json();
                if (data.error) throw new Error(data.error);
                if (actionType === 'convert') {
                    if (data.format === 'pdf') {
                        document.getElementById('imgConvert').style.display = 'none';
                        document.getElementById('pdfIcon').style.display = 'block';
                    } else {
                        document.getElementById('imgConvert').src = data.image;
                        document.getElementById('imgConvert').style.display = 'inline-block';
                        document.getElementById('pdfIcon').style.display = 'none';
                    }
                    document.getElementById('statsConvert').innerText = `Final Size: ${(data.size / 1024).toFixed(1)} KB`;
                    document.getElementById('resConvert').style.display = 'block';
                    document.getElementById('dlConvertBtn').onclick = () => downloadData(data.image, 'converted', data.format);
                } else {
                    document.getElementById('imgCompress').src = data.image;
                    document.getElementById('statsCompress').innerText = data.message;
                    document.getElementById('resCompress').style.display = 'block';
                    document.getElementById('dlCompressBtn').onclick = () => downloadData(data.image, 'compressed', 'jpg');
                }
            } catch (err) {
                alert('Server Error: ' + err.message);
            } finally {
                hideLoading();
            }
        }
